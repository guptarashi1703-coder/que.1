import java.io.*;
import java.util.*;
class UserMainCode{
public int isPalinNum(int input1){
int rev=0,original=input1;
    while(input1>0){
        rev=rev*10+input1%10;
        input1=input1/10;
    }if( rev==original)
    return 2;
    else
    return 1;
    }
}
