Que.25 Non-Repeated Digits Count
Ans.25
import java.io.*;
import java.util.*;
class UserMainCode{
public int nonRepeatDigitsCount(int input1){
int freq[]=new int[10];
while(input1>0){
int digit=input1%10;
freq[digit]++;
input1/=10;
}int count=0;
for(int i=0;i<10;i++){
if(freq[i]==1){
count++;
}
}
return count;
}
}
