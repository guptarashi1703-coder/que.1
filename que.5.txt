import java.io.*;
import java.util.*
class UserMainCode{
public int addLastDigits(int input1,int input2){
if(input1<0){
input1=-input1;
}if(input2<0){
input2=-input2;
}int lastDigitinput1=input1%10;
int lastDigitinput2=input2%10;
return lastDigitinput1 + lastDigitinput2;
}
}
